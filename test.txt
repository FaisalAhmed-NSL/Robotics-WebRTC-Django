def initConnection(portNo, baudRate):
    try:
        ser = serial.Serial(portNo, baudRate)
        print("Device Connected")
        return ser
    except serial.SerialException:
        print("Failed to connect to device")

def sendData(se, data, digits):
    myString = "$"
    for d in data:
        myString += str(d).zfill(digits)
    try:
        se.write(myString.encode())
        print(myString)
    except serial.SerialException:
        print("Failed to transmit data")

def direction(d):
    if d in ARDUINO_COMMANDS:
        command = ARDUINO_COMMANDS[d]
        print(d)
        sendData(ser, command, 2)
    else:
        print('Invalid direction')

def on_connect(client, userdata, flags, rc):
    if rc == 0:
        print(f"Connected to broker at {client._host}:{client._port}")
        client.subscribe(MQTT_TOPIC)
    else:
        sendData(ser, ARDUINO_COMMANDS["stop"], 2)
        print(f"Connection to broker at {client._host}:{client._port} failed with error code {rc}")
        print("Trying to connect again in 5 seconds")
        time.sleep(5)
        client.reconnect()

def on_disconnect(client, userdata, rc):
    if rc != 0:
        sendData(ser, ARDUINO_COMMANDS["stop"], 2)
        print(f"Disconnected from broker at {client._host}:{client._port}. Trying to reconnect...")
        client.reconnect()

def on_message(client, userdata, message):
    print("Received message:", message.payload)
    try:
        payload = json.loads(message.payload.decode('utf-8'))
        direction(payload['direction'])
    except (json.JSONDecodeError, KeyError):
        direction(message.payload.decode('utf-8').strip())

# Try to connect to the MQTT broker
connected = False
while not connected:
    try:
        # Use context manager for serial communication
        with serial.Serial("/dev/ttyACM0", 9600) as ser:
            # Initialize MQTT client and connect to broker
            client = mqtt.Client()
            client.on_connect = on_connect
            client.on_disconnect = on_disconnect
            client.on_message = on_message
            client.connect("172.16.16.54", 1883, 60)

            # Start the main loop
            client.loop_forever()

            sendData(ser, ARDUINO_COMMANDS["stop"], 2)
            # Close the serial connection
            ser.close()
        connected = True
